                    <h2 class="poem-title">The Rare Ones</h2>
                    <div class="poem-content">in a world too loud to listen, i realized something... some people are like treasure ‚Äî rare, hidden, sometimes buried deep beneath noise and distraction.

it's hard to find someone who really understands you, someone who shares that quiet spark that makes you feel seen, like they've touched the part of you you couldn't even name.

my favorite people are the ones who evolve ‚Äî open-minded, ready to learn, ready to live new experiences.

and when you find that sweet chemistry, that unexplainable click between souls... the magic begins. but even magic fades when one soul stops showing up. so sometimes, you have to let things go.

because if they truly know your worth, they'll circle back ‚Äî and if they don't... well, we live in an age where meaning is vanishing, where connection's become just another notification.

everyone's online ‚Äî but it feels like no one's there. scrolling, replying, talking ‚Äî but not really speaking.

i'm the type of guy who craves the real ‚Äî a small talk that feels big, a shared silence that says everything, eye contact that speaks louder than words.

        const config = {
            heroPoems: [
                "If you have to ask for it, it loses its value",
                "Don't chase butterflies, plant a garden instead",
                "Each one a step, each one a mirror",
                "At least we are watching the same moon",
                "Life will always be hard. But you get to choose your hard",
                "In the end, presence is the rarest treasure left"
            ],
            typewriterSpeed: 90
        };

        function typeWriter(element, text, speed = 90) {
            let i = 0;
            element.textContent = '';
            function type() {
                if (i < text.length) {
                    element.textContent += text.charAt(i);
                    i++;
                    setTimeout(type, speed);
                }
            }
            type();
        }

        function showPage(pageId) {
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            const targetPage = document.getElementById(pageId + '-page');
            if (targetPage) targetPage.classList.add('active');
            
            document.querySelectorAll('.nav-links a').forEach(link => link.classList.remove('active'));
            document.querySelector(`[data-page="${pageId}"]`)?.classList.add('active');
            
            if (pageId === 'home') {
                setTimeout(initHeroAnimation, 300);
            }
            window.scrollTo(0, 0);
        }

        function initMobileMenu() {
            const toggle = document.getElementById('mobileMenuToggle');
            const sidebar = document.getElementById('sidebar');
            
            if (!toggle || !sidebar) return;
            
            toggle.addEventListener('click', () => {
                toggle.classList.toggle('active');
                sidebar.classList.toggle('active');
            });
            
            document.addEventListener('click', (e) => {
                if (!sidebar.contains(e.target) && !toggle.contains(e.target) && sidebar.classList.contains('active')) {
                    sidebar.classList.remove('active');
                    toggle.classList.remove('active');
                }
            });
        }

        function initHeroAnimation() {
            const heroPoem = document.getElementById('heroPoem');
            if (!heroPoem) return;
            const randomPoem = config.heroPoems[Math.floor(Math.random() * config.heroPoems.length)];
            typeWriter(heroPoem, '"' + randomPoem + '"', config.typewriterSpeed);
        }

        function initNewsletterForm() {
            const form = document.getElementById('newsletterForm');
            if (!form) return;
            
            form.addEventListener('submit', (e) => {
                e.preventDefault();
                const button = form.querySelector('button');
                const originalText = button.textContent;
                
                button.textContent = 'Thank you!';
                setTimeout(() => {
                    form.reset();
                    button.textContent = originalText;
                }, 3000);
            });
        }

        function initNavigation() {
            document.querySelectorAll('.nav-links a').forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const page = link.dataset.page;
                    if (page) showPage(page);
                });
            });
            
            document.querySelectorAll('.prose-teaser-button').forEach(button => {
                button.addEventListener('click', () => {
                    showPage(button.dataset.page);
                });
            });
            
            document.querySelectorAll('.read-more').forEach(button => {
                button.addEventListener('click', () => {
                    const poemId = button.dataset.poem;
                    showPage('collection');
                    setTimeout(() => {
                        const targetPoem = document.getElementById(poemId);
                        if (targetPoem) {
                            targetPoem.scrollIntoView({ behavior: 'smooth', block: 'start' });
                            targetPoem.style.boxShadow = '0 0 30px rgba(122, 155, 142, 0.25)';
                            setTimeout(() => {
                                targetPoem.style.boxShadow = '';
                            }, 2500);
                        }
                    }, 400);
                });
            });
            
            document.querySelectorAll('.favorite-item').forEach(item => {
                const poemId = item.dataset.poem;
                if (poemId) {
                    item.addEventListener('click', () => {
                        showPage('collection');
                        setTimeout(() => {
                            const targetPoem = document.getElementById(poemId);
                            if (targetPoem) {
                                targetPoem.scrollIntoView({ behavior: 'smooth', block: 'start' });
                                targetPoem.style.boxShadow = '0 0 30px rgba(122, 155, 142, 0.25)';
                                setTimeout(() => {
                                    targetPoem.style.boxShadow = '';
                                }, 2500);
                            }
                        }, 400);
                    });
                }
            });
        }

        function initScrollAnimations() {
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.style.opacity = '1';
                        entry.target.style.transform = 'translateY(0)';
                    }
                });
            }, { threshold: 0.1 });
            
            document.querySelectorAll('.poem-card').forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(30px)';
                card.style.transition = `all 0.8s cubic-bezier(0.4, 0, 0.2, 1) ${index * 0.15}s`;
                observer.observe(card);
            });
            
            document.querySelectorAll('.full-poem, .prose-piece').forEach((poem, index) => {
                poem.style.opacity = '0';
                poem.style.transform = 'translateY(40px)';
                poem.style.transition = `all 1s cubic-bezier(0.4, 0, 0.2, 1) ${index * 0.2}s`;
                observer.observe(poem);
            });
        }

        function init() {
            initMobileMenu();
            initHeroAnimation();
            initNewsletterForm();
            initNavigation();
            initScrollAnimations();
        }

        if (document.readyState === 'loading') {

        } else {
            init();
        }
